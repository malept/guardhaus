37/// Represents a `username` (or user hash, if the header's `userhash` parameter is `true`).
38#[derive(Clone, Debug, PartialEq)]
39pub enum Username {
40    /// Either an ASCII-encoded username, or a userhash (if the header's `userhash` parameter is
41    /// `true`).
42    Plain(String),
43    /// An RFC 5987-encoded username.
44    Encoded(ExtendedValue),
45}
46
47impl fmt::Display for Username {
48    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
49        match *self {
50            Username::Plain(ref username) => write!(f, "{}", username),
51            Username::Encoded(ref encoded) => write!(f, "{}", encoded),
52        }
53    }
54}

56/// Parameters for the `Authorization` header when using the `Digest` scheme.
57///
58/// The parameters are described in more detail in
59/// [RFC 7616](https://tools.ietf.org/html/rfc7616#section-3.4).
60/// Unless otherwise noted, the parameter name maps to the struct variable name.
61#[derive(Clone, PartialEq, Debug)]
62pub struct Digest {
63    /// Either the user name or the user hash (if `userhash` is `true` - see [RFC 7616, section
64    /// 3.4.4](https://tools.ietf.org/html/rfc7616#section-3.4.4)).
65    pub username: Username,
66    /// Authentication realm.
67    pub realm: String,
68    /// Cryptographic nonce.
69    pub nonce: String,
70    /// Nonce count, parameter name `nc`. Optional only in RFC 2067 mode.
71    pub nonce_count: Option<NonceCount>,
72    /// The hexadecimal digest of the payload as described by the RFCs.
73    pub response: String,
74    /// Either the absolute path or URI of the HTTP request, parameter name `uri`.
75    pub request_uri: String,
76    /// The hash algorithm to use when generating the `response`.
77    pub algorithm: HashAlgorithm,
78    /// Quality of protection. Optional only in RFC 2067 mode.
79    pub qop: Option<Qop>,
80    /// Cryptographic nonce from the client. Optional only in RFC 2067 mode.
81    pub client_nonce: Option<String>,
82    /// Optional opaque string.
83    pub opaque: Option<String>,
84    /// The character set to use when generating the A1 value or the userhash. Added for RFC 7616.
85    pub charset: Option<Charset>,
86    /// Whether `username` is a userhash. Added for RFC 7616.
87    pub userhash: bool,
88}

90macro_rules! ensure_ok {
91    ($expr: expr) => {
92        match $expr {
93            Ok(value) => value,
94            Err(_) => return Err(headers::Error::invalid()),
95        }
96    };
97}

99impl Credentials for Digest {
100    const SCHEME: &'static str = "Digest";
101
102    fn decode(value: &HeaderValue) -> Option<Self> {
103        if let Ok(serialized) = value.to_str() {
104            match serialized.parse() {
105                Ok(digest) => Some(digest),
106                Err(_) => None,
107            }
108        } else {
109            None
110        }
111    }
112
113    fn encode(&self) -> headers::HeaderValue {
114        let mut parameters = DigestParameters::new();
115        match self.username {
116            Username::Plain(ref username) => parameters.append("username", username, true),
117            Username::Encoded(ref encoded) => {
118                parameters.append("username*", &encoded.to_string(), false)
119            }
120        }
121        parameters.append("realm", &self.realm, true);
122        parameters.append("nonce", &self.nonce, true);
123        if let Some(ref nonce_count) = self.nonce_count {
124            parameters.append("nc", &nonce_count.to_string(), false);
125        }
126        parameters.append("response", &self.response, true);
127        parameters.append("uri", &self.request_uri, true);
128        parameters.append("algorithm", &self.algorithm.to_string(), false);
129        if let Some(ref qop) = self.qop {
130            parameters.append("qop", &qop.to_string(), false);
131        }
132        if let Some(ref client_nonce) = self.client_nonce {
133            parameters.append("cnonce", client_nonce, true);
134        }
135        if let Some(ref opaque) = self.opaque {
136            parameters.append("opaque", opaque, true);
137        }
138        if let Some(ref charset) = self.charset {
139            parameters.append("charset", &charset.to_string(), false);
140        }
141        if self.userhash {
142            parameters.append("userhash", "true", false);
143        }
144        HeaderValue::from_str(&parameters.to_string())
145            .expect("Could not generate HeaderValue for Authorization")
146    }
147}

149fn parse_username(map: &HashMap<UniCase<String>, String>) -> Result<Username, Error> {
150    if let Some(value) = unraveled_map_value(map, "username") {
151        if unraveled_map_value(map, "username*").is_some() {
152            Err(Error::invalid())
153        } else {
154            Ok(Username::Plain(value))
155        }
156    } else if let Some(encoded) = unraveled_map_value(map, "username*") {
157        if let Some(userhash) = unraveled_map_value(map, "userhash") {
158            if userhash == "true" {
159                return Err(Error::invalid());
160            }
161        }
162
163        let extended_value = ensure_ok!(encoded.parse());
164        Ok(Username::Encoded(extended_value))
165    } else {
166        Err(Error::invalid())
167    }
168}
169
170macro_rules! unravel_map_value {
171    ($map: ident, $param_name: literal) => {
172        match unraveled_map_value(&$map, $param_name) {
173            Some(value) => value,
174            None => return Err(Error::invalid()),
175        }
176    };
177}

179impl FromStr for Digest {
180    type Err = Error;
181    fn from_str(s: &str) -> Result<Digest, Error> {
182        let param_map = parse_parameters(s);
183        let username: Username = parse_username(&param_map)?;
184        let realm: String = unravel_map_value!(param_map, "realm");
185        let nonce: String = unravel_map_value!(param_map, "nonce");
186        let nonce_count = ensure_ok!(NonceCount::from_parameters(&param_map));
187        let response: String = unravel_map_value!(param_map, "response");
188        let request_uri: String = unravel_map_value!(param_map, "uri");
189        let algorithm: HashAlgorithm =
190            if let Some(value) = unraveled_map_value(&param_map, "algorithm") {
191                ensure_ok!(HashAlgorithm::from_str(&value[..]))
192            } else {
193                HashAlgorithm::Md5
194            };
195        let charset: Option<Charset> =
196            if let Some(value) = unraveled_map_value(&param_map, "charset") {
197                let utf8 = UniCase::new("utf-8".to_owned());
198                if UniCase::new(value) == utf8 {
199                    Some(Charset::UTF_8)
200                } else {
201                    return Err(Error::invalid());
202                }
203            } else {
204                None
205            };
206        let userhash: bool = if let Some(value) = unraveled_map_value(&param_map, "userhash") {
207            match &value[..] {
208                "true" => true,
209                "false" => false,
210                _ => return Err(Error::invalid()),
211            }
212        } else {
213            false
214        };
215        let qop = ensure_ok!(Qop::from_parameters(&param_map));
216        Ok(Digest {
217            username,
218            realm,
219            nonce,
220            nonce_count,
221            response,
222            request_uri,
223            algorithm,
224            qop,
225            client_nonce: unraveled_map_value(&param_map, "cnonce"),
226            opaque: unraveled_map_value(&param_map, "opaque"),
227            charset,
228            userhash,
229        })
230    }
231}

233impl Digest {
234    /// Returns a copy wrapped with an Authorization header.
235    pub fn to_header(&self) -> Authorization<Digest> {
236        Authorization(self.clone())
237    }

239    /// Generates a userhash, as defined in
240    /// [RFC 7616, section 3.4.4](https://tools.ietf.org/html/rfc7616#section-3.4.4).
241    pub fn userhash(algorithm: &HashAlgorithm, username: Vec<u8>, realm: String) -> String {
242        let mut to_hash = username.clone();
243        to_hash.push(b':');
244        to_hash.append(&mut realm.into_bytes());
245        println!("value: {:?}", std::str::from_utf8(&to_hash));
246        algorithm.hex_digest(to_hash.as_slice())
247    }
248
249    /// Validates a userhash (as defined in
250    /// [RFC 7616, section 3.4.4](https://tools.ietf.org/html/rfc7616#section-3.4.4)), given a
251    /// `Digest` header.
252    ///
253    /// If userhash is `false`, returns `false`.
254    pub fn validate_userhash(&self, username: Username) -> bool {
255        match self.username {
256            Username::Plain(ref userhash) => {
257                let name = match username {
258                    Username::Plain(value) => value.into_bytes(),
259                    Username::Encoded(encoded) => encoded.value,
260                };
261                *userhash == Digest::userhash(&self.algorithm, name, self.realm.clone())
262            }
263            Username::Encoded(_) => false,
264        }
265    }
266
267    fn simple_a1(username: Username, realm: String, password: String) -> Vec<u8> {
268        let mut a1: Vec<u8> = match username {
269            Username::Plain(name) => name.into_bytes(),
270            Username::Encoded(encoded) => encoded.value,
271        };
272        a1.push(b':');
273        a1.append(&mut realm.into_bytes());
274        a1.push(b':');
275        a1.append(&mut password.into_bytes());
276
277        a1
278    }
279
280    /// Generates a simple hexadecimal digest from an A1 value and given algorithm.
281    ///
282    /// This is intended to be used in applications that use the `htdigest` style of secret hash
283    /// generation.
284    ///
285    /// To see how a simple A1 value is constructed, see
286    /// [RFC 7616, section 3.4.2](https://tools.ietf.org/html/rfc7616#section-3.4.2).
287    /// This is the definition when the algorithm is "unspecified".
288    pub fn simple_hashed_a1(
289        algorithm: &HashAlgorithm,
290        username: Username,
291        realm: String,
292        password: String,
293    ) -> String {
294        algorithm.hex_digest(Digest::simple_a1(username, realm, password).as_slice())
295    }
296
297    // RFC 7616, Section 3.4.2
298    fn a1(&self, username: Username, password: String) -> Result<Vec<u8>, Error> {
299        let realm = self.realm.clone();
300        match self.algorithm {
301            HashAlgorithm::Md5 | HashAlgorithm::Sha256 | HashAlgorithm::Sha512256 => {
302                Ok(Digest::simple_a1(username, realm, password))
303            }
304
305            HashAlgorithm::Md5Session
306            | HashAlgorithm::Sha256Session
307            | HashAlgorithm::Sha512256Session => {
308                if let Some(ref client_nonce) = self.client_nonce {
309                    let simple_hashed_a1 = self
310                        .algorithm
311                        .hex_digest(Digest::simple_a1(username, realm, password).as_slice());
312                    let mut a1 = simple_hashed_a1.into_bytes();
313                    a1.push(b':');
314                    a1.append(&mut self.nonce.clone().into_bytes());
315                    a1.push(b':');
316                    a1.append(&mut client_nonce.clone().into_bytes());
317                    Ok(a1)
318                } else {
319                    Err(Error::invalid())
320                }
321            }
322        }
323    }
324
325    /// Generates a hexadecimal digest from an A1 value.
326    ///
327    /// To see how an A1 value is constructed, see
328    /// [RFC 7616, section 3.4.2](https://tools.ietf.org/html/rfc7616#section-3.4.2).
329    fn hashed_a1(&self, username: Username, password: String) -> Result<String, Error> {
330        let a1 = ensure_ok!(self.a1(username, password));
331        Ok(self.algorithm.hex_digest(a1.as_slice()))
332    }
333
334    // RFC 7616, Section 3.4.3
335    fn a2(&self, method: Method, entity_body: &[u8]) -> String {
336        match self.qop {
337            Some(Qop::AuthInt) => format!(
338                "{}:{}:{}",
339                method,
340                self.request_uri,
341                self.algorithm.hex_digest(entity_body)
342            ),
343            _ => format!("{}:{}", method, self.request_uri),
344        }
345    }
346
347    fn hashed_a2(&self, method: Method, entity_body: &[u8]) -> String {
348        self.algorithm
349            .hex_digest(self.a2(method, entity_body).as_bytes())
350    }
351
352    fn kd(algorithm: &HashAlgorithm, secret: String, data: String) -> String {
353        let value = format!("{}:{}", secret, data);
354        algorithm.hex_digest(value.as_bytes())
355    }
356
357    fn using_username_and_password(
358        &self,
359        method: Method,
360        entity_body: &[u8],
361        username: Username,
362        password: String,
363    ) -> Result<String, Error> {
364        let a1 = ensure_ok!(self.hashed_a1(username, password));
365        self.using_hashed_a1(method, entity_body, a1)
366    }

368    /// Generates a digest, given an HTTP request and a password.
369    ///
370    /// `entity_body` is defined in
371    /// [RFC 2616, secion 7.2](https://tools.ietf.org/html/rfc2616#section-7.2).
372    pub fn using_password(
373        &self,
374        method: Method,
375        entity_body: &[u8],
376        password: String,
377    ) -> Result<String, Error> {
378        let a1 = ensure_ok!(self.hashed_a1(self.username.clone(), password));
379        self.using_hashed_a1(method, entity_body, a1)
380    }
381
382    /// Generates a digest, given an HTTP request and a hexadecimal digest of an A1 string.
383    ///
384    /// `entity_body` is defined in
385    /// [RFC 2616, secion 7.2](https://tools.ietf.org/html/rfc2616#section-7.2).
386    ///
387    /// This is intended to be used in applications that use the `htdigest` style of secret hash
388    /// generation.
389    pub fn using_hashed_a1(
390        &self,
391        method: Method,
392        entity_body: &[u8],
393        a1: String,
394    ) -> Result<String, Error> {
395        let a2 = self.hashed_a2(method, entity_body);
396        let data: String;
397        if let Some(ref qop) = self.qop {
398            match *qop {
399                Qop::Auth | Qop::AuthInt => {
400                    if self.client_nonce.is_none() || self.nonce_count.is_none() {
401                        return Err(Error::invalid());
402                    }
403                    let nonce = self.nonce.clone();
404                    let nonce_count = self.nonce_count.clone().expect("No nonce count found");
405                    let client_nonce = self.client_nonce.clone().expect("No client nonce found");
406                    data = format!("{}:{}:{}:{}:{}", nonce, nonce_count, client_nonce, qop, a2);
407                }
408            }
409        } else {
410            data = format!("{}:{}", self.nonce, a2);
411        }
412        Ok(Digest::kd(&self.algorithm, a1, data))
413    }

415    fn validate_using_username_and_password(
416        &self,
417        method: Method,
418        entity_body: &[u8],
419        username: Username,
420        password: String,
421    ) -> bool {
422        if let Ok(hex_digest) =
423            self.using_username_and_password(method, entity_body, username, password)
424        {
425            hex_digest == self.response
426        } else {
427            false
428        }
429    }

431    /// Validates a `Digest.response`, given an HTTP request and a password.
432    ///
433    /// `entity_body` is defined in
434    /// [RFC 2616, secion 7.2](https://tools.ietf.org/html/rfc2616#section-7.2).
435    pub fn validate_using_password(
436        &self,
437        method: Method,
438        entity_body: &[u8],
439        password: String,

441        self.validate_using_username_and_password(
442            method,
443            entity_body,
444            self.username.clone(),
445            password,

449    /// Validates a `Digest.username` and `Digest.response`, given an HTTP request, a username,
450    /// and a password. If a userhash is specified, that is validated first.
451    ///
452    /// `entity_body` is defined in
453    /// [RFC 2616, secion 7.2](https://tools.ietf.org/html/rfc2616#section-7.2).
454    pub fn validate_using_userhash_and_password(
455        &self,
456        method: Method,
457        entity_body: &[u8],
458        username: Username,
459        password: String,
460    ) -> bool {
461        if self.userhash && !self.validate_userhash(username.clone()) {
462            return false;
463        }
464        self.validate_using_username_and_password(method, entity_body, username, password)
465    }

467    /// Validates a `Digest.response`, given an HTTP request and a hexadecimal digest of an
468    /// A1 string.
469    ///
470    /// `entity_body` is defined in
471    /// [RFC 2616, secion 7.2](https://tools.ietf.org/html/rfc2616#section-7.2).
472    ///
473    /// This is intended to be used in applications that use the `htdigest` style of secret hash
474    /// generation.
475    pub fn validate_using_hashed_a1(&self, method: Method, entity_body: &[u8], a1: String) -> bool {
476        if let Ok(hex_digest) = self.using_hashed_a1(method, entity_body, a1) {
477            hex_digest == self.response
478        } else {
479            false
480        }
481    }
