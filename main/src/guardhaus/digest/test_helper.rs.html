30fn serialize_headers(headers: headers::HeaderMap) -> Result<String, http::header::ToStrError> {
31    let mut serialized = String::new();
32    for (name, value) in headers.iter() {
33        serialized.push_str(name.as_str());
34        serialized.push_str(": ");
35        serialized.push_str(value.to_str()?);
36        serialized.push('\r');
37        serialized.push('\n');
38    }
39
40    Ok(serialized)
41}

43pub fn assert_parsed_header_equal(expected: Digest, data: &str) {
44    assert!(data.starts_with("Digest "));
45    let digest_params = data
46        .get(7..)
47        .expect("Header value should be at least 7 chars");
48    let actual: Digest = digest_params
49        .parse()
50        .expect("Could not parse digest params");
51    assert_eq!(expected, actual);
52}

58pub fn assert_serialized_header_equal(digest: Digest, actual: &str) {
59    let mut headers = headers::HeaderMap::new();
60    let credentials = format!(
61        "Digest {}",
62        digest
63            .encode()
64            .to_str()
65            .expect("Could not serialize Digest credentials")
66    );
67    headers.insert(
68        http::header::AUTHORIZATION,
69        HeaderValue::from_str(credentials.as_str())
70            .expect("Could not deserialize Digest credentials"),
71    );
72    let expected = serialize_headers(headers).expect("Could not serialize headers");
73    assert_eq!(expected, format!("{}\r\n", actual))
74}

76pub fn parse_digest_header(data: &str) -> Digest {
77    assert!(data.starts_with("Digest "));
78    let digest_params = data
79        .get(7..)
80        .expect("Header value should be at least 7 chars");
81    let header_value = HeaderValue::from_str(digest_params).expect("Could not parse digest header");
82    match Digest::decode(&header_value) {
83        Some(digest) => digest,
84        None => panic!("Could not decode header into Digest struct"),
85    }
86}
