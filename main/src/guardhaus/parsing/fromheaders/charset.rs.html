176impl Charset {
177    fn name(&self) -> &str {
178        match &self.0 {
179            Charset_::Us_Ascii => "US-ASCII",
180            Charset_::Iso_8859_1 => "ISO-8859-1",
181            Charset_::Iso_8859_2 => "ISO-8859-2",
182            Charset_::Iso_8859_3 => "ISO-8859-3",
183            Charset_::Iso_8859_4 => "ISO-8859-4",
184            Charset_::Iso_8859_5 => "ISO-8859-5",
185            Charset_::Iso_8859_6 => "ISO-8859-6",
186            Charset_::Iso_8859_7 => "ISO-8859-7",
187            Charset_::Iso_8859_8 => "ISO-8859-8",
188            Charset_::Iso_8859_9 => "ISO-8859-9",
189            Charset_::Iso_8859_10 => "ISO-8859-10",
190            Charset_::Shift_Jis => "Shift-JIS",
191            Charset_::Euc_Jp => "EUC-JP",
192            Charset_::Iso_2022_Kr => "ISO-2022-KR",
193            Charset_::Euc_Kr => "EUC-KR",
194            Charset_::Iso_2022_Jp => "ISO-2022-JP",
195            Charset_::Iso_2022_Jp_2 => "ISO-2022-JP-2",
196            Charset_::Iso_8859_6_E => "ISO-8859-6-E",
197            Charset_::Iso_8859_6_I => "ISO-8859-6-I",
198            Charset_::Iso_8859_8_E => "ISO-8859-8-E",
199            Charset_::Iso_8859_8_I => "ISO-8859-8-I",
200            Charset_::Gb2312 => "GB2312",
201            Charset_::Big5 => "5",
202            Charset_::Koi8_R => "KOI8-R",
203            Charset_::Utf_8 => "UTF-8",
204            Charset_::Ext(value) => value.as_str(),
205            Charset_::_Unknown => unreachable!("Charset::_Unknown"),
206        }
207    }
208}

210impl fmt::Display for Charset {
211    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
212        f.write_str(self.name())
213    }

216#[derive(Debug)]
217pub struct CharsetFromStrError(());
218
219impl FromStr for Charset {
220    type Err = CharsetFromStrError;
221    fn from_str(s: &str) -> Result<Self, Self::Err> {
222        Ok(Charset(match s.to_ascii_uppercase().as_ref() {
223            "US-ASCII" => Charset_::Us_Ascii,
224            "ISO-8859-1" => Charset_::Iso_8859_1,
225            "ISO-8859-2" => Charset_::Iso_8859_2,
226            "ISO-8859-3" => Charset_::Iso_8859_3,
227            "ISO-8859-4" => Charset_::Iso_8859_4,
228            "ISO-8859-5" => Charset_::Iso_8859_5,
229            "ISO-8859-6" => Charset_::Iso_8859_6,
230            "ISO-8859-7" => Charset_::Iso_8859_7,
231            "ISO-8859-8" => Charset_::Iso_8859_8,
232            "ISO-8859-9" => Charset_::Iso_8859_9,
233            "ISO-8859-10" => Charset_::Iso_8859_10,
234            "SHIFT-JIS" => Charset_::Shift_Jis,
235            "EUC-JP" => Charset_::Euc_Jp,
236            "ISO-2022-KR" => Charset_::Iso_2022_Kr,
237            "EUC-KR" => Charset_::Euc_Kr,
238            "ISO-2022-JP" => Charset_::Iso_2022_Jp,
239            "ISO-2022-JP-2" => Charset_::Iso_2022_Jp_2,
240            "ISO-8859-6-E" => Charset_::Iso_8859_6_E,
241            "ISO-8859-6-I" => Charset_::Iso_8859_6_I,
242            "ISO-8859-8-E" => Charset_::Iso_8859_8_E,
243            "ISO-8859-8-I" => Charset_::Iso_8859_8_I,
244            "GB2312" => Charset_::Gb2312,
245            "5" => Charset_::Big5,
246            "KOI8-R" => Charset_::Koi8_R,
247            "UTF-8" => Charset_::Utf_8,
248            _unknown => return Err(CharsetFromStrError(())),
249        }))
250    }
251}

253#[test]
254fn test_parse() {
255    assert_eq!(Charset::US_ASCII, "us-ascii".parse().unwrap());
256    assert_eq!(Charset::US_ASCII, "US-Ascii".parse().unwrap());
257    assert_eq!(Charset::US_ASCII, "US-ASCII".parse().unwrap());
258    assert_eq!(Charset::SHIFT_JIS, "Shift-JIS".parse().unwrap());
259    assert_eq!(Charset::UTF_8, "UTF-8".parse().unwrap());
260    assert!("abcd".parse::<Charset>().is_err());
261}
