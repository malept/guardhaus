31/// Represents a parameter of a Digest Authorization header.
32struct DigestParameter {
33    key: String,
34    value: String,
35    quoted: bool,
36}
37
38/// Serializes digest header parameters into a string.
39pub struct DigestParameters(Vec<DigestParameter>);
40
41impl DigestParameters {
42    /// Creates a new SerializedParameters builder.
43    pub fn new() -> Self {
44        Self(vec![])
45    }
46
47    /// Append a header parameter to a serialized header.
48    pub fn append(&mut self, key: &str, value: &str, quoted: bool) {
49        self.0.push(DigestParameter {
50            key: key.to_string(),
51            value: value.to_string(),
52            quoted,
53        });
54    }
55}
56
57impl std::fmt::Display for DigestParameters {
58    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
59        let mut first_element = true;
60        for param in self.0.iter() {
61            if first_element {
62                first_element = false;
63            } else {
64                write!(f, ", ")?;
65            }
66            write!(f, "{}=", param.key)?;
67            if param.quoted {
68                write!(f, "\"")?;
69            }
70            write!(f, "{}", param.value)?;
71            if param.quoted {
72                write!(f, "\"")?;
73            }
74        }
75        Ok(())
76    }
77}

79pub fn parse_parameters(s: &str) -> HashMap<UniCase<String>, String> {
80    let parameters: Vec<String> =
81        from_comma_delimited(s).expect("Could not parse header parameters");
82    let mut param_map: HashMap<UniCase<String>, String> = HashMap::with_capacity(parameters.len());
83    for parameter in parameters {
84        let parts: Vec<&str> = parameter.splitn(2, '=').collect();
85        param_map.insert(
86            UniCase::new(parts[0].trim().to_owned()),
87            parts[1].trim().trim_matches('"').to_owned(),
88        );
89    }
90
91    param_map
92}

94pub fn unraveled_map_value(map: &HashMap<UniCase<String>, String>, key: &str) -> Option<String> {
95    let value = map.get(&UniCase::new(key.to_owned()))?;
96    match percent_decode(value.as_bytes()).decode_utf8() {
97        Ok(string) => Some(string.into_owned()),
98        Err(_) => None,
99    }
