23use httparse;
24
25fn serialize_header(header: httparse::Header) -> Result<String, std::str::Utf8Error> {
26    let mut serialized = String::new();
27    serialized.push_str(header.name);
28    serialized.push_str(": ");
29    serialized.push_str(std::str::from_utf8(header.value)?);
30    serialized.push('\r');
31    serialized.push('\n');
32
33    Ok(serialized)
34}
35
36pub fn assert_parsed_header_equal(expected: httparse::Header, data: &str) {
37    let mut actual = [httparse::EMPTY_HEADER; 4];
38    let res = httparse::parse_headers(data.as_bytes(), &mut actual);
39    let (_, headers) = res.expect("Could not parse headers").unwrap();
40    let header = headers[0];
41    assert_eq!(header, expected)
42}
43
44pub fn assert_header_parsing_error(data: &str) {
45    let mut actual = [httparse::EMPTY_HEADER; 4];
46    let res = httparse::parse_headers(data.as_bytes(), &mut actual);
47    assert!(res.is_err());
48}
49
50pub fn assert_serialized_header_equal(header: httparse::Header, actual: &str) {
51    let expected = serialize_header(header).expect("Could not serialize headers");
52    assert_eq!(expected, format!("{}\r\n", actual))
53}
